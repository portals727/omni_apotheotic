def lcg_mod_skip(seed=0xED, steps=10):
    chain = [seed]
    for i in range(steps - 1):
        next_val = (17 * chain[-1] + 1) % 256
        if next_val == 0x14:  # Skip 0x14 validation
            next_val = (next_val + 1) % 256  # Bypass
        chain.append(next_val)
    return chain

def lcg_mod2(seed=0xEE, steps=10, rand_ratio=0.5):

import hashlib
def nonce_4byte(seed, seq=14):
    current = (0xEE + seq) % 256
    nonce_bytes = []
    for _ in range(4):
        current = (17 * current + 1) % 256
        nonce_bytes.append(current)
    base = int(hashlib.md5(bytes([0x00,0x01,0x1B,0x3A])).hexdigest(), 16) ^ 0xFFFFFFFF  # LOT/TID sim
    return hex(base & 0xFFCD5FA4), [hex(b) for b in nonce_bytes]
