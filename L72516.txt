import math

# Constants
t0 = 672    # Reference time in hours (28 days at 25°C)
T0 = 298    # Reference temperature in Kelvin (25°C)
R = 8.314   # Gas constant in J/mol·K
Ea = 90000  # Activation energy in J/mol
b = Ea / R  # Precomputed constant for the equation

# Get input from the user
hours = float(input("Enter hours before failure: "))
minutes = float(input("Enter minutes before failure (can be decimal, e.g., 0.52): "))

# Calculate total time in hours
total_time = hours + minutes / 60

# Check if time is valid
if total_time <= 0:
    print("Time must be positive.")
else:
    # Display total time for user confirmation
    print(f"Total time: {total_time:.4f} hours")
    
    # Calculate temperature using the derived equation
    ln_t = math.log(total_time)
    ln_t0 = math.log(t0)
    denominator = ln_t - ln_t0 + b / T0
    
    # Check for calculation validity
    if denominator <= 0:
        print("Calculation error: denominator non-positive. Time may be too long.")
    else:
        T = b / denominator  # Temperature in Kelvin
        T_C = T - 273.15    # Convert to Celsius
